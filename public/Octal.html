Octal

การเเปลงเลขฐาน 10 เป็นฐาน 8

Coding to C program

                    
        
        #include < stdio.h >

        int main() {
            int dec;
            int oct[100];
            int i = 0;
            int j;

            printf("Enter decimal number ");
            scanf_s("%d", &dec);

            do {
                oct[i] = dec % 8;
                dec /= 8;
                i++;
            } while (dec > 0);
                printf("\n\nOctal is ");
                for (j = i - 1; j >= 0; j--) {
                printf("%d", oct[j]);
            }
            return 0;
        }
