Binary

การเเปลงเลขฐาน 10 เป็นฐาน 2

Coding to C program

                    
    #include < stdio.h >

    int main() {

        int dec;
        int b[100];
        int i = 0; 
        int j;

        printf("Enter decimal number ");
        scanf_s("%d", &dec);

        do {
            b[i] = dec % 2;
            dec /= 2;
            i++;
        } while (dec > 0);
        printf("\n\nBinary is ");
        for (j = i - 1; j >= 0; j--) {
            printf("%d", b[j]);
        }
        return 0;
    }

เป็นขั้นตอนการเเปลงเลขฐาน
5) จะเริ่มจาก b[i] = dec % 2; นำข้อมูลที่เก็บไว้ใน dec มาหาร 2 เเละนำเก็บไว้เศษใน b และจะไม่เปลี่ยนค่าข้อมูลลของ dec เเละจะได้เลขเฉพาะ 1 เเละ 0 เท่านัน
6) dec /= 2; นำข้อมูลที่เก็บไว้ใน dec มาหาร 2 พร้อมทั้งเเทนค่าผลลัพท์ลงที่ได้มีลงใน dec เเละเป็นตัวที่จะทำให้ค่าของเรานั้นมีเศษเเค่ 1 เเละ 0
7) i++; บวกจำนวนครั้งลงใน i
8) เช็คเงื่อนไขของ dec ว่าค่านั้นน้อยกว่า 0 แล้วหรือยัง ด้วย while (dec > 0); หากน้อยกว่าเเล้วการทำงานจะหลุดออกจาก loop ทันที
