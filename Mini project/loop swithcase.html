Swith Case

บอกโรคต่างๆที่คนไข้เป็น ตามที่กำหนด

Codding to C program

                    

        #include 


        int main(void) {


	        int x;
	        printf("set your Value 1-6: ");
	        scanf_s("%d", &x);

	        switch (x) {

	        case 1:
		        printf("case is heart disease\n");
		        /*โรคหัวใจ*/
		        break;

	        case 2:
		        printf("case is diabetes\n");
		        /*โรคเบาหวาน*/
		        break;

	        case 3:
		        printf("case is osteoporosis\n");
		        /*โรคกระดูกพรุน*/
		        break;

	        case 4:
		        printf("case is gastritis\n");
		        /*โรคกระเพาะอาหารอักเสบ*/
		        break;

	        case 5:
		        printf("case is depression\n");
		        /*ภาวะซึมเศร้า*/
		        break;

	        case 6:
		        printf("case is allergy\n");
		        /*โรคภูมิเเพ้*/
		        break;

            default :
                printf("case is can't find\n");
                /*หาไม่เจอ*/
                break;
                }
		return 0;
	    }

swithcase นั้นจะทำงานตามเงื่อนไขของตัวเเปลที่เรากำหนด โดยจะเเสดงค่าออกตามปลิ้น F ของเเต่ล่ะ case
โดยเราต้องกำหนด case เเต่ล่ะ case ด้วยตัวเลขลงท้ายข้างหลัง เพื่อให้ switch เลือก case ที่ตรงกับค่าข้อมูลใน x 

                    
            #include < stdio.h >
