Loop While

การหาค่า เกรดเฉลี่ยเเบบง่าย

Codding to C program

                    
    #include < stdio.h >

    int main() {
                            
        int counter;
        int grade;
        int total;
        int average;
                            
        total = 0;
        counter = 1;
                            
        while (counter <= 10) {                            
            printf("Enter grade: ");
            scanf_s("%d", &grade);
            total = total + grade;
            counter = counter + 1;
        }
                            
        average = total / 10;
        printf("class average is %d\n", average);
        return 0;
    }

โดย loop while นั้นเป็นการใช้คำสั่งทำซ้ำโดยการกำหนดเงื่อนไขเป็นครั้งๆ
หากเงื่อไขยังเป็นจริงอยู่ก็จะทำไปเรื่อยๆ เเต่หากเงื่อไขเป็นเท็จเเล้วจะหลุดออกจาก loop ทันที่ 

                    
            #include < stdio.h >

#include คือคำสั่งการใช้ function ของ library < stdio.h > 

                    
            int counter;
            int grade;
            int total;
            int average;

เป็นการกำหนดตัวเเปลเพื่อเก็บค่าของข้อมูล 

                    
            total = 0;
            counter = 1;

กำหนดค่าข้อมูลลงในตัวเเปล 

                    
            while (counter <= 10) {                            
                .....
            }

ตั้งค่าเงื่อนไขให้ loop มีการทำซ้ำ น้อยกว่าเท่ากับ10 ครั้ง

ภายใน loop เงื่อนไขตัวนี้เราเรียกว่า expression 

                    
            printf("Enter grade: ");
            scanf_s("%d", &grade);
            total = total + grade;
            counter = counter + 1;

printf คือการเเสดงผลขึ้นหน้าจอ ในที่นี้คือบอกให้ใส่ เกรด

scanf_s เป็นการป้อนค่าลงไปโดยจะบรรจุค่าลงใน grade จำนวน 1 ตัว ทั้งหมด 10 ครั้ง

%d คือ ค่าตัวเลขจำนวนเต็ม

total จะถูกตั้งค่าเริ่มต้นที่ 0 เเล้วนำมาบวกกับ grade ได้ค่าเท่าไรจะเเทนลงไปจากค่าเดิมที่เป็น 0 เพื่อนำไปบวกต่อไป 

                    
            average = total / 10;
            printf("class average is %d\n", average);
            return 0;

นำค่าที่ได้จากการนำ total ไปบวกกับ grade มาหาร 10

printf("... %d\n", average) เป็นคำสั่งเเสดงค่าของ average โดยค่านั้นเป็นจำนวนเต็ม

return 0; คำสั่งการรีเทริน์ค่ากลับไปยัง library เพื่อจบการทำงาน
